Changing the currentColor using the color style property:

    <style>
        g { color: darkcyan;}
        rect {fill: currentColor}
        text {fill: currentColor}
        line {stroke: currentColor}
    <style>
    <script>
    d3.select("g")
      .transition().delay(1000)
      .style("color", "magenta")
          .transition().delay(1000)
          .style("color", "olive")
              .transition().delay(1000)
              .style("color", "orange")
    </script>
