Use the JavaScript console to experiment with them using D3

<script>
    const numArray = [9,2,12,9,71];
    const strArray = ['9','2','12','9','71','Y',null];

    const tasks = [
        {task: 'Debugging', cost: 1.40, priority: 1, date: new Date(2023,2,15)},
        {task: 'Reviewing', cost: 0.60, priority: 2, date: new Date(2023,2,10)},
        {task: 'Editing', cost: 0.25, priority: 5, date: new Date(2023,3,9)},
        {task: 'Testing', cost: 0.10, priority: 1, date: new Date(2023,1,27)},
    ];

    const taskmap = new Map(tasks.map(e => [e.task, e]));
    const numset = new Set(numArray);
    const str = "953zyx2";
</script>

Open console to see the results of the following commands

Examples with d3.least()

d3.least(numArray)
d3.least(numArray, (a,b) => a - b)
d3.least(strArray)
d3.least(strArray, (a,b) => a - b)
d3.least(tasks, (a,b) => a.cost - b.cost)
d3.least(tasks, (a,b) => a.priority - b.priority)

Examples with d3.greatest()

d3.greatest(numArray)
d3.greatest(numArray, (a,b) => a - b)
d3.greatest(strArray)
d3.greatest(strArray, (a,b) => a - b)
d3.greatest(tasks, (a,b) => a.cost - b.cost)
d3.greatest(tasks, (a,b) => a.priority - b.priority)

Examples with d3.leastIndex() and d3.greatestIndex()

d3.leastIndex(numArray)
d3.leastIndex(tasks, (a,b) => a.date - b.date)
d3.greatestIndex(numArray)
d3.greatestIndex(strArray)
d3.greatestIndex(tasks, d => d.cost)

Examples with d3.ascending() and d3.descending()

numArray.sort((a,b) => d3.descending(a,b))
numArray.sort(d3.descending)
tasks.sort((a,b) => d3.ascending(a.cost, b.cost))

Examples with d3.rank()

d3.rank(tasks)
d3.rank(tasks, d => d.date)
d3.rank(tasks, (a,b) => d3.ascending(a.date, b.date))

Examples with d3.quickselect()

d3.quickselect([9,6,3,2,8,7,5],3)
d3.quickselect([9,6,3,2,8,7,5],3,1,5,d3.descending)
